/*
 * 图的遍历
 */

#include<bits/stdc++.h>  //头文件头文件头文件
//#include<vector>
//#include<queue>
//#include<algorithm>
#define MAXN 100005
using namespace std;

int n, m;
vector<int> p[MAXN];

queue<int> q;
bool visited[MAXN];
//bool u2[MAXN];

void dfs(int x){
    cout<<x<<' ';
    for (int i = 0,size = p[x].size(); i < size; ++i) {
        if(!visited[p[x][i]]){
            visited[p[x][i]] = true;
            dfs(p[x][i]);
        }
    }
}

void bfs(int x){
    memset(visited, false, sizeof(visited));
    visited[x] = true;
    q.push(x);
    while (!q.empty()){
        int x = q.front();
        q.pop();
        cout<<x<<' ';
        for(int i = 0,size = p[x].size();i < size;i++){
            if(!visited[p[x][i]]){
                visited[p[x][i]] = true;
                q.push(p[x][i]);
            }
        }
    }
}

int main(){

    cin >> n >> m;
    for (int i = 1; i <= m ; ++i) {
        int x,y;
        cin >> x >> y;
        p[x].push_back(y); //用邻接表记录下文献x有参考文献y
    }

    for (int i=1; i<=n; i++) sort(p[i].begin(), p[i].end());//标准vector排序
    visited[1] = true;
    dfs(1);
    cout<<endl;
    bfs(1);
    cout<<endl;


}

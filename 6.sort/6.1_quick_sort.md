void qsort(int l,int r)//自己写的快排
{
    int mid=a[(l+r)/2];//找中间的数进行2分
    int i=l,j=r;
    do{
        while(a[i]<mid)
            i++;//找左半部分大于中间数的
        while(a[j]>mid)
            j--;//找右半部分小于中间数的
        if(i<=j)
        {
            swap(a[i],a[j]);//换位
            i++;//左指针右移
            j--;//右指针左移
        }
    }
    while(i<=j);
    if(l<j) qsort(l,j);
    if(i<r) qsort(i,r);
}

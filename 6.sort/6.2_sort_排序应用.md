* 统计数组中元素个数

  std::count(a, a + N, 0) << std::endl;

```C++

int N;
int num[1011];
int cnt = 0;

int main(){
    cin >> N;
    int temp;
    memset(num,0,sizeof num);
    for(int i = 1; i <=N; i ++){
        scanf("%d",&temp);
        num[temp]++;
    }

    cout <<1000-std::count(num+1, num + 1001, 0)<<endl ;
    for(int i = 1; i <= 1000; i ++){

        if(num[i]) { printf("%d ", i); }
    }
}
```
